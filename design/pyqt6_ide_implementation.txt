class SettingsDialog(QDialog):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.setWindowTitle("IDE Settings")
        self.setFixedSize(500, 400)
        self.init_ui()

    def init_ui(self):
        layout = QVBoxLayout(self)
        tabs = QTabWidget()

        # Tab 1: API Configuration
        api_tab = QWidget()
        api_layout = QFormLayout(api_tab)
        
        self.openai_key = QLineEdit()
        self.openai_key.setEchoMode(QLineEdit.EchoMode.Password)
        api_layout.addRow("OpenAI Key:", self.openai_key)
        
        self.anthropic_key = QLineEdit()
        self.anthropic_key.setEchoMode(QLineEdit.EchoMode.Password)
        api_layout.addRow("Anthropic Key:", self.anthropic_key)

        self.gemini_key = QLineEdit()
        self.gemini_key.setEchoMode(QLineEdit.EchoMode.Password)
        api_layout.addRow("Google Gemini Key:", self.gemini_key)

        tabs.addTab(api_tab, "API Configurations")

        # Tab 2: General Settings
        gen_tab = QWidget()
        gen_layout = QFormLayout(gen_tab)
        gen_layout.addRow("Local Repository:", QLineEdit("/path/to/repo"))
        gen_layout.addRow("Model Path:", QLineEdit("/usr/local/models"))
        tabs.addTab(gen_tab, "General")

        layout.addWidget(tabs)
        
        btns = QHBoxLayout()
        save_btn = QPushButton("Save")
        cancel_btn = QPushButton("Cancel")
        save_btn.clicked.connect(self.accept)
        cancel_btn.clicked.connect(self.reject)
        btns.addStretch()
        btns.addWidget(save_btn)
        btns.addWidget(cancel_btn)
        layout.addLayout(btns)

class NexusDashboard(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Nexus IDE | Multi-Agent Collaborative Environment")
        self.resize(1200, 800)
        self.setStyleSheet(DARK_STYLESHEET)
        self.init_ui()

    def init_ui(self):
        # Central Widget & Main Vertical Splitter (Top/Bottom)
        central_widget = QWidget()
        self.setCentralWidget(central_widget)
        main_layout = QVBoxLayout(central_widget)
        
        v_splitter = QSplitter(Qt.Orientation.Vertical)
        h_splitter = QSplitter(Qt.Orientation.Horizontal)

        # 1. Left Panel: File Tree
        self.file_model = QFileSystemModel()
        self.file_model.setRootPath(QDir.currentPath())
        self.tree = QTreeView()
        self.tree.setModel(self.file_model)
        self.tree.setRootIndex(self.file_model.index(QDir.currentPath()))
        self.tree.setContextMenuPolicy(Qt.ContextMenuPolicy.CustomContextMenu)
        self.tree.customContextMenuRequested.connect(self.show_tree_context_menu)
        h_splitter.addWidget(self.tree)

        # 2. Center Panel: Code Editor
        self.editor = QTextEdit()
        self.editor.setPlaceholderText("Select a file to view code...")
        self.editor.setFont(QFont("JetBrains Mono", 10))
        h_splitter.addWidget(self.editor)

        # 3. Right Panel: Agent Management
        agent_panel = QWidget()
        agent_layout = QVBoxLayout(agent_panel)
        agent_layout.addWidget(QLabel("<b>Agent Allocation</b>"))
        self.agent_list = QListWidget()
        self.agent_list.addItem("Claude 3.5 - Frontend (Idle)")
        self.agent_list.addItem("GPT-4 - Backend (Running)")
        self.agent_list.addItem("Llama 3 - Test Suite (Waiting)")
        agent_layout.addWidget(self.agent_list)
        h_splitter.addWidget(agent_panel)

        h_splitter.setStretchFactor(1, 2)
        v_splitter.addWidget(h_splitter)

        # 4. Bottom Panel: Log Console
        self.log_console = QTextEdit()
        self.log_console.setReadOnly(True)
        self.log_console.setStyleSheet("background-color: #0c0c0c; color: #00ff00; font-family: 'Consolas';")
        self.log_console.append("[SYSTEM] Initializing Nexus Orchestrator...")
        self.log_console.append("[EVENT_BUS] Handshake with Agent 'Claude-3.5-Sonnet' established.")
        v_splitter.addWidget(self.log_console)
        
        v_splitter.setStretchFactor(0, 3)
        main_layout.addWidget(v_splitter)

        # Menu & Status Bar
        self.setStatusBar(QStatusBar())
        self.statusBar().showMessage("System Ready | API: Connected")
        
        menubar = self.menuBar()
        file_menu = menubar.addMenu("File")
        settings_action = QAction("Settings", self)
        settings_action.triggered.connect(self.open_settings)
        file_menu.addAction(settings_action)

    def open_settings(self):
        dialog = SettingsDialog(self)
        dialog.exec()

    def show_tree_context_menu(self, position):
        menu = QMenu()
        assign_menu = menu.addMenu("Assign Agent")
        assign_menu.addAction("Assign to Claude 3.5 (Frontend)")
        assign_menu.addAction("Assign to GPT-4 (Backend)")
        assign_menu.addAction("Assign to Gemini (Testing)")
        menu.exec(self.tree.viewport().mapToGlobal(position))

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = NexusDashboard()
    window.show()
    sys.exit(app.exec())
